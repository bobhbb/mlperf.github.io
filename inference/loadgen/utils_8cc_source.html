38 std::string DateTimeString(const char* format,
39  std::chrono::system_clock::time_point tp,
40  bool append_ms) {
41  std::time_t tp_time_t = std::chrono::system_clock::to_time_t(tp);
42  std::tm date_time = *std::localtime(&tp_time_t);
43  constexpr size_t kDateTimeMaxSize = 256;
44  char date_time_cstring[kDateTimeMaxSize];
45  std::strftime(date_time_cstring, kDateTimeMaxSize, format, &date_time);
46  std::string date_time_string(date_time_cstring);
47  if (!append_ms) {
48  return date_time_string;
49  }
50 
51  auto tp_time_t_part = std::chrono::system_clock::from_time_t(tp_time_t);
52  auto tp_remainder = tp - tp_time_t_part;
53  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(tp_remainder)
54  .count();
55  if (ms < 0 || ms >= 1000) {
56  LogDetail([ms](AsyncDetail& detail) {
57  detail("WARNING: Unexpected milliseconds getting date and time.", "ms",
58  ms);
59  });
60  }
61  std::string ms_string = std::to_string(ms);
62  // Prefix with zeros so length is always 3.
63  ms_string.insert(0, std::min<size_t>(2, 3 - ms_string.length()), '0');
64  return date_time_string + "." + ms_string;
65 }
