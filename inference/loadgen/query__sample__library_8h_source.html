29 /// \brief The interface a client implements to coordinate with the loadgen
30 /// which samples should be loaded.
31 class QuerySampleLibrary {
32  public:
33  virtual ~QuerySampleLibrary() {}
34 
35  /// \brief A human readable name for the model.
36  virtual const std::string& Name() const = 0;
37 
38  /// \brief Total number of samples in library.
39  virtual size_t TotalSampleCount() = 0;
40 
41  /// \brief The number of samples that are guaranteed to fit in RAM.
42  virtual size_t PerformanceSampleCount() = 0;
43 
44  /// \brief Loads the requested query samples into memory.
45  /// \details Paired with calls to UnloadSamplesFromRam.
46  /// In the MultiStream scenarios:
47  /// * Samples will appear more than once.
48  /// * SystemUnderTest::IssueQuery will only be called with a set of samples
49  /// that are neighbors in the vector of samples here, which helps
50  /// SUTs that need the queries to be contiguous.
51  /// In all other scenarios:
52  /// * A previously loaded sample will not be loaded again.
53  virtual void LoadSamplesToRam(
54  const std::vector<QuerySampleIndex>& samples) = 0;
55 
56  /// \brief Unloads the requested query samples from memory.
57  /// \details In the MultiStream scenarios:
58  /// * Samples may be unloaded the same number of times they were loaded;
59  /// however, if the implementation de-dups loaded samples rather than
60  /// loading samples into contiguous memory, it may unload a sample the
61  /// first time they see it unloaded without a refcounting scheme, ignoring
62  /// subsequent unloads. A refcounting scheme would also work, but is not
63  /// a requirement.
64  /// In all other scenarios:
65  /// * A previously unloaded sample will not be unloaded again.
66  virtual void UnloadSamplesFromRam(
67  const std::vector<QuerySampleIndex>& samples) = 0;
68 };
