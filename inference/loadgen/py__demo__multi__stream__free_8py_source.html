39 # Processes queries in 3 slices that complete at different times.
40 def process_query_async(query_samples, i_slice):
41  time.sleep(.001 * (i_slice + 1))
42  responses = []
43  samples_to_complete = query_samples[i_slice:len(query_samples):3]
44  for s in samples_to_complete:
45  responses.append(mlperf_loadgen.QuerySampleResponse(s.id, 0, 0))
46  mlperf_loadgen.QuerySamplesComplete(responses)

49 def issue_query(query_samples):
50  threading.Thread(target=process_query_async,
51  args=(query_samples, 0)).start()
52  threading.Thread(target=process_query_async,
53  args=(query_samples, 1)).start()
54  threading.Thread(target=process_query_async,
55  args=(query_samples, 2)).start()
