37 ///
38 /// \enum TestScenario
39 /// * **SingleStream**
40 /// + Issues queries containing a single sample.
41 /// + The next query is only issued once the previous one has completed.
42 /// + Internal LoadGen latency between queries is not included in the
43 /// latency results.
44 /// + **Final performance result is:** the 90th percentile latency.
45 /// * **MultiStream**
46 /// + Attempts to issue queries containing N samples each at a uniform rate.
47 /// - N is specified by \link
48 /// mlperf::TestSettings::multi_stream_samples_per_query
49 /// multi_stream_samples_per_query \endlink.
50 /// - The rate is specified by \link
51 /// mlperf::TestSettings::multi_stream_target_qps multi_stream_target_qps
52 /// \endlink.
53 /// + The loadgen will skip sending for one interval if the SUT falls behind
54 /// too much.
55 /// + By default, only a single query may be outstanding at a time.
56 /// + Latency is tracked on a per-sample basis, as opposed to per-query.
57 /// + The samples of each query are guaranteed to be contiguous with respect
58 /// to the order they were loaded in the QuerySampleLibrary.
59 /// + **Final performance result is:** PASS if the 90 percentile latency is
60 /// under a given threshold. FAIL otherwise.
61 /// - Threshold is specified by \link
62 /// mlperf::TestSettings::multi_stream_target_latency_ns
63 /// multi_stream_target_latency_ns \endlink.
64 /// * **MultiStreamFree**
65 /// + Behaves similar to MultiStream, with the exceptions that it:
66 /// - Allows up to N async queries where N is limited only by the latency
67 /// target.
68 /// - Issues queries at a variable rate corresponding to when the N'th
69 /// oldest query completes.
70 /// + Not an official MLPerf scenario, but is maintained for evaluation
71 /// and testing purposes.
72 /// + Compared to MultiStream, there is no frequency quantization, which
73 /// allows the results to reflect small performance improvements.
74 /// + **Final performance result is:** samples per second.
75 /// * **Server**
76 /// + Sends queries with a single sample.
77 /// + Queries have a random poisson (non-uniform) arrival rate that, when
78 /// averaged, hits the target QPS.
79 /// + **Final performance result is:** PASS if the 90 percentile latency is
80 /// under a given threshold. FAIL otherwise.
81 /// - Threshold is specified by \link
82 /// mlperf::TestSettings::server_target_latency_ns server_target_latency_ns
83 /// \endlink.
84 /// * **Offline**
85 /// + Sends all N samples to the SUT inside of a single query.
86 /// + The samples of the query are guaranteed to be contiguous with respect
87 /// to the order they were loaded in the QuerySampleLibrary.
88 /// (WIP. Not true yet.)
89 /// + **Final performance result is:** samples per second.
90 ///
91 enum class TestScenario {
92  SingleStream,
93  MultiStream,
94  MultiStreamFree,
95  Server,
96  Offline,
97 };

99 ///
100 /// \enum TestMode
101 /// * **SubmissionRun**
102 /// + Runs accuracy mode followed by performance mode.
103 /// + TODO: Implement further requirements as decided by MLPerf.
104 /// * **AccuracyOnly**
105 /// + Runs each sample from the QSL through the SUT a least once.
106 /// + Outputs responses to an accuracy json that can be parsed by a model +
107 /// sample library specific script.
108 /// * **PerformanceOnly**
109 /// + Runs the performance traffic for the given scenario, as described in
110 /// the comments for TestScenario.
111 /// * **FindPeakPerformance**
112 /// + TODO: Not implemented yet.
113 /// + Determines the maximumum QPS for the Server scenario.
114 /// + Determines the maximum samples per query for the MultiStream and
115 /// MultiStreamFree scenarios.
116 /// + Not applicable for SingleStream or Offline scenarios.
117 ///
118 enum class TestMode {
119  SubmissionRun,
120  AccuracyOnly,
121  PerformanceOnly,
122  FindPeakPerformance,
123 };

138  uint64_t single_stream_expected_latency_ns = 1000000;

140 
141  // ==================================
142  /// \name MultiStream-specific
143  /**@{*/
144  /// \brief The uniform rate at which queries are produced.
145  /// Does not apply to the MultiStreamFree scenario.
146  double multi_stream_target_qps = 10.0;
147  /// \brief The latency constraint for the MultiStream scenario.
148  uint64_t multi_stream_target_latency_ns = 100000000;
149  /// \brief The number of samples in each query.
150  int multi_stream_samples_per_query = 4;
151  /// \brief The maximum number of queries, to which a SUT has not responded,
152  /// before the loadgen will throttle issuance of new queries.
153  int multi_stream_max_async_queries = 1;
154  /**@}*/
155 
156  // ==================================
157  /// \name Server-specific
158  /**@{*/
159  /// \brief The average QPS of the poisson distribution.
160  /// Only used as a hint in SearchForPeakPerformance.
161  double server_target_qps = 1;
162  /// \brief The latency constraint for the Server scenario.
163  uint64_t server_target_latency_ns = 100000000;
164  /// \brief TODO: Implement this.
165  bool server_coalesce_queries = false;
166  /**@}*/
167 
168  // ==================================
169  /// \name Offline-specific
170  /**@{*/
171  /// \brief Specifies the QPS the SUT expects to hit for the offline load.
172  /// The loadgen generates 10% more queries than it thinks it
173  /// needs to meet the minimum test duration.
174  double offline_expected_qps = 1;
175  /**@}*/
176 
177  // ==================================
178  /// \name Test duration
179  /// The test runs until **both** min duration and min query count have been
180  /// met. However, it will exit before that point if **either** max duration or
181  /// max query count have been reached.
182  /**@{*/
183  uint64_t min_duration_ms = 10000;
184  uint64_t max_duration_ms = 0; ///< 0: Infinity.
185  uint64_t min_query_count = 100;
186  uint64_t max_query_count = 0; ///< 0: Infinity.
187  /**@}*/
188 
189  // ==================================
190  /// \name Random number generation
191  /// There are 3 separate seeds, so each dimension can be changed
192  /// independently.
193  /**@{*/
194  /// \brief Affects which subset of samples from the QSL are chosen for
195  /// the performance sample set and accuracy sample sets.
196  uint64_t qsl_rng_seed = 0;
197  /// \brief Affects the order in which samples from the performance set will
198  /// be included in queries.
199  uint64_t sample_index_rng_seed = 0;
200  /// \brief Affects the poisson arrival process of the Server scenario.
201  /// \details Different seeds will appear to "jitter" the queries
202  /// differently in time, but should not affect the average issued QPS.
203  uint64_t schedule_rng_seed = 0;
204  /**@}*/
205 };
