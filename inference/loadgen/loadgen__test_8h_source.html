121  /// \brief Registers a test at runtime.
122  template <typename TestF, typename... Args>
123  static void Register(const char* name, TestF test, Args&&... args) {
124  std::function<void()> test_closure =
125  std::bind(test, std::forward<Args>(args)...);
126  tests().insert({std::move(name), std::move(test_closure)});
127  }

129  /// \brief Runs all currently registered tests that match the given filter.
130  static int Run(std::function<bool(const char*)> filter) {
131  // Determine which tests are enabled.
132  std::vector<TestMap::value_type*> enabled_tests;
133  for (auto& test : tests()) {
134  if (filter(test.first)) {
135  enabled_tests.push_back(&test);
136  }
137  }
138  const size_t enabled = enabled_tests.size();
139  std::cout << enabled << " of " << tests().size() << " tests enabled.\n";
140 
141  // Run the tests.
142  std::vector<const char*> failures;
143  for (size_t i = 0; i < enabled; i++) {
144  const char* name = enabled_tests[i]->first;
145  std::cout << "[" << (i + 1) << "/" << enabled << "] : " << name << " : ";
146  std::cout.flush();
147  test_fails() = 0;
148  try {
149  enabled_tests[i]->second(); // Run the test.
150  } catch (std::exception& e) {
151  constexpr bool TestThrewException = true;
152  FAIL_IF(TestThrewException) && FAIL_EXP(e.what());
153  }
154  if (test_fails() > 0) {
155  failures.push_back(name);
156  std::cerr << "\n FAILED: " << name << "\n";
157  } else {
158  std::cout << "SUCCESS\n";
159  }
160  }
161 
162  // Summarize.
163  if (enabled_tests.empty()) {
164  std::cerr << "Check your test filter.\n";
165  } else if (failures.empty()) {
166  std::cout << "All " << enabled << " tests passed! \\o/\n";
167  } else {
168  std::cout << failures.size() << " of " << enabled << " tests failed:\n";
169  for (auto failed_test_name : failures) {
170  std::cout << " " << failed_test_name << "\n";
171  }
172  }
173  return failures.size();
174  }
