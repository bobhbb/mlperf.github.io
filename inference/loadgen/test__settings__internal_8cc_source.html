26 TestSettingsInternal::TestSettingsInternal(
27  const TestSettings &requested_settings, size_t qsl_performance_sample_count)
28  : requested(requested_settings),
29  scenario(requested.scenario),
30  mode(requested.mode),
31  samples_per_query(1),
32  target_qps(1),
33  max_async_queries(0),
34  target_duration(std::chrono::milliseconds(requested.min_duration_ms)),
35  min_duration(std::chrono::milliseconds(requested.min_duration_ms)),
36  max_duration(std::chrono::milliseconds(requested.max_duration_ms)),
37  min_query_count(requested.min_query_count),
38  max_query_count(requested.max_query_count),
39  min_sample_count(0),
40  qsl_rng_seed(requested.qsl_rng_seed),
41  sample_index_rng_seed(requested.sample_index_rng_seed),
42  schedule_rng_seed(requested.schedule_rng_seed),
43  accuracy_log_rng_seed(requested.accuracy_log_rng_seed),
44  accuracy_log_probability(requested.accuracy_log_probability),
45  print_timestamps(requested.print_timestamps),
46  performance_issue_unique(requested.performance_issue_unique),
47  performance_issue_same(requested.performance_issue_same),
48  performance_issue_same_index(requested.performance_issue_same_index),
49  performance_sample_count(0) {
50  // Target QPS, target latency, and max_async_queries.
51  switch (requested.scenario) {
52  case TestScenario::SingleStream:
53  target_qps = static_cast<double>(std::nano::den) /
54  requested.single_stream_expected_latency_ns;
55  max_async_queries = 1;
56  target_latency_percentile =
57  requested.single_stream_target_latency_percentile;
58  break;
59  case TestScenario::MultiStream: {
60  max_async_queries = requested.multi_stream_max_async_queries;
61  target_qps = requested.multi_stream_target_qps;
62  double target_latency_seconds =
63  max_async_queries / requested.multi_stream_target_qps;
64  target_latency =
65  SecondsToDuration<std::chrono::nanoseconds>(target_latency_seconds);
66  target_latency_percentile =
67  requested.multi_stream_target_latency_percentile;
68  break;
69  }
70  case TestScenario::MultiStreamFree:
71  max_async_queries = requested.multi_stream_max_async_queries;
72  target_qps = requested.multi_stream_target_qps;
73  target_latency =
74  std::chrono::nanoseconds(requested.multi_stream_target_latency_ns);
75  target_latency_percentile =
76  requested.multi_stream_target_latency_percentile;
77  break;
78  case TestScenario::Server:
79  if (requested.server_target_qps >= 0.0) {
80  target_qps = requested.server_target_qps;
81  } else {
82  LogDetail([server_target_qps = requested.server_target_qps,
83  target_qps = target_qps](AsyncDetail &detail) {
84  detail.Error("Invalid value for server_target_qps requested.",
85  "requested", server_target_qps, "using", target_qps);
86  });
87  }
88  target_latency =
89  std::chrono::nanoseconds(requested.server_target_latency_ns);
90  target_latency_percentile = requested.server_target_latency_percentile;
91  max_async_queries = requested.server_max_async_queries;
92  break;
93  case TestScenario::Offline:
94  if (requested.offline_expected_qps >= 0.0) {
95  target_qps = requested.offline_expected_qps;
96  } else {
97  LogDetail([offline_expected_qps = requested.offline_expected_qps,
98  target_qps = target_qps](AsyncDetail &detail) {
99  detail.Error("Invalid value for offline_expected_qps requested.",
100  "requested", offline_expected_qps, "using", target_qps);
101  });
102  }
103  max_async_queries = 1;
104  break;
105  }
106 
107  // Performance Sample Count: TestSettings override QSL ->
108  // PerformanceSampleCount
109  performance_sample_count = (requested.performance_sample_count_override == 0)
110  ? qsl_performance_sample_count
111  : requested.performance_sample_count_override;
112 
113  // Samples per query.
114  if (requested.scenario == TestScenario::MultiStream ||
115  requested.scenario == TestScenario::MultiStreamFree) {
116  samples_per_query = requested.multi_stream_samples_per_query;
117  }
118 
119  // In the offline scenario, coalesce all queries into a single query.
120  if (requested.scenario == TestScenario::Offline) {
121  // TODO: Should the spec require a max duration for large query counts?
122  // kSlack is used to make sure we generate enough samples for the SUT
123  // to take longer than than the minimum test duration required by the
124  // MLPerf spec.
125  constexpr double kSlack = 1.1;
126  int target_sample_count =
127  kSlack * DurationToSeconds(target_duration) * target_qps;
128  samples_per_query =
129  (requested.performance_issue_unique || requested.performance_issue_same)
130  ? performance_sample_count
131  : std::max<int>(min_query_count, target_sample_count);
132  min_query_count = 1;
133  target_duration = std::chrono::milliseconds(0);
134  }
135 
136  min_sample_count = min_query_count * samples_per_query;
137 
138  // Validate TestSettings
139  if (requested.performance_issue_same &&
140  (requested.performance_issue_same_index >= performance_sample_count)) {
141  LogDetail(
142  [performance_issue_same_index = requested.performance_issue_same_index,
143  performance_sample_count =
144  performance_sample_count](AsyncDetail &detail) {
145  detail.Error(
146  "Sample Idx to be repeated in performance_issue_same mode"
147  " cannot be greater than loaded performance_sample_count.",
148  "performance_issue_same_index", performance_issue_same_index,
149  "performance_sample_count", performance_sample_count);
150  });
151  }
152 
153  if (requested.performance_issue_unique && requested.performance_issue_same) {
154  LogDetail([performance_issue_unique = requested.performance_issue_unique,
155  performance_issue_same =
156  requested.performance_issue_same](AsyncDetail &detail) {
157  detail.Error(
158  "Performance_issue_unique and performance_issue_same, both"
159  " cannot be true at the same time.",
160  "performance_issue_unique", performance_issue_unique,
161  "performance_issue_same", performance_issue_same);
162  });
163  }
164 }

167  switch (scenario) {
168  case TestScenario::SingleStream:
169  return "Single Stream";
170  case TestScenario::MultiStream:
171  return "Multi Stream";
172  case TestScenario::MultiStreamFree:

322 /// \todo The TestSettings::FromConfig definition belongs in a test_settings.cc
323 /// file which doesn't yet exist. To avoid churn so close to the submission
324 /// deadline, adding a test_settings.cc file has been deferred to v0.6.
325 int TestSettings::FromConfig(const std::string &path, const std::string &model,
326  const std::string &scenario) {
327  // TODO: move this method to a new file test_settings.cc
328  std::map<std::string, std::string> kv;
329 
330  // lookup key/value pairs from config
331  auto lookupkv = [&](const std::string &model, const std::string &scenario,
332  const std::string &key, uint64_t *val_l, double *val_d,
333  double multiplier = 1.0) {
334  std::map<std::string, std::string>::iterator it;
335  std::string found;
336  // lookup exact key first
337  it = kv.find(model + "." + scenario + "." + key);
338  if (it != kv.end()) {
339  found = it->second;
340  } else {
341  // lookup key with model wildcard
342  it = kv.find("*." + scenario + "." + key);
343  if (it != kv.end()) {
344  found = it->second;
345  } else {
346  return false;
347  }
348  }
349  // if we get here, found will be set
350  if (val_l) {
351  *val_l =
352  strtoul(found.c_str(), nullptr, 0) * static_cast<int>(multiplier);
353  }
354  if (val_d) *val_d = strtod(found.c_str(), nullptr) * multiplier;
355  return true;
356  };
357 
358  // dirt simple config parser
359  std::ifstream fss(path);
360  std::string line;
361  int line_nr = 0;
362  int errors = 0;
363  if (!fss.is_open()) {
364  LogDetail([p = path](AsyncDetail &detail) {
365  detail.Error("can't open file ", p);
366  });
367  return -ENOENT;
368  }
369  while (std::getline(fss, line)) {
370  line_nr++;
371  std::istringstream iss(line);
372  std::string s, k;
373  int looking_for = 0; // 0=key, 1=equal, 2=value
374  while (iss >> s) {
375  if (s == "#" && looking_for != 2) {
376  // done with this line
377  break;
378  }
379  if (looking_for == 2) {
380  // got key and value
381  const char *start = s.c_str();
382  char *stop;
383  (void)strtoul(start, &stop, 0);
384  if (start + s.size() == stop) {
385  kv[k] = s;
386  continue;
387  }
388  (void)strtod(start, &stop);
389  if (start + s.size() == stop) {
390  kv[k] = s;
391  continue;
392  }
393  errors++;
394  LogDetail([l = line_nr](AsyncDetail &detail) {
395  detail.Error("value needs to be integer or double, line=", l);
396  });
397  break;
398  }
399  if (looking_for == 1 && s != "=") {
400  errors++;
401  LogDetail([l = line_nr](AsyncDetail &detail) {
402  detail.Error("expected 'key=value', line=", l);
403  });
404  break;
405  }
406  if (looking_for == 0) k = s;
407  looking_for++;
408  }
409  }
410  if (errors != 0) return -EINVAL;
411 
412  uint64_t val;
413 
414  // keys that apply to all scenarios
415  if (lookupkv(model, scenario, "mode", &val, nullptr)) {
416  switch (val) {
417  case 0:
418  mode = TestMode::SubmissionRun;
419  break;
420  case 1:
421  mode = TestMode::AccuracyOnly;
422  break;
423  case 2:
424  mode = TestMode::PerformanceOnly;
425  break;
426  case 3:
427  mode = TestMode::FindPeakPerformance;
428  break;
429  default:
430  LogDetail([](AsyncDetail &detail) {
431  detail.Error("Invalid value passed to Mode key in config.");
432  });
433  break;
434  }
435  }
436  lookupkv(model, scenario, "min_duration", &min_duration_ms, nullptr);
437  lookupkv(model, scenario, "max_duration", &max_duration_ms, nullptr);
438  lookupkv(model, scenario, "min_query_count", &min_query_count, nullptr);
439  lookupkv(model, scenario, "max_query_count", &max_query_count, nullptr);
440  lookupkv(model, scenario, "qsl_rng_seed", &qsl_rng_seed, nullptr);
441  lookupkv(model, scenario, "sample_index_rng_seed", &sample_index_rng_seed,
442  nullptr);
443  lookupkv(model, scenario, "schedule_rng_seed", &schedule_rng_seed, nullptr);
444  lookupkv(model, scenario, "accuracy_log_rng_seed", &accuracy_log_rng_seed,
445  nullptr);
446  lookupkv(model, scenario, "accuracy_log_probability", nullptr,
447  &accuracy_log_probability, 0.01);
448  if (lookupkv(model, scenario, "print_timestamps", &val, nullptr))
449  print_timestamps = (val == 0) ? false : true;
450  if (lookupkv(model, scenario, "performance_issue_unique", &val, nullptr))
451  performance_issue_unique = (val == 0) ? false : true;
452  if (lookupkv(model, scenario, "performance_issue_same", &val, nullptr))
453  performance_issue_same = (val == 0) ? false : true;
454  lookupkv(model, scenario, "performance_issue_same_index",
455  &performance_issue_same_index, nullptr);
456  lookupkv(model, scenario, "performance_sample_count_override",
457  &performance_sample_count_override, nullptr);
458 
459  // keys that apply to SingleStream
460  lookupkv(model, "SingleStream", "target_latency_percentile", nullptr,
461  &single_stream_target_latency_percentile, 0.01);
462  lookupkv(model, "SingleStream", "target_latency",
463  &single_stream_expected_latency_ns, nullptr, 1000 * 1000);
464 
465  // keys that apply to MultiStream
466  lookupkv(model, "MultiStream", "target_latency_percentile", nullptr,
467  &multi_stream_target_latency_percentile, 0.01);
468  lookupkv(model, "MultiStream", "target_qps", nullptr,
469  &multi_stream_target_qps);
470  if (lookupkv(model, "MultiStream", "samples_per_query", &val, nullptr))
471  multi_stream_samples_per_query = static_cast<int>(val);
472  if (lookupkv(model, "MultiStream", "max_async_queries", &val, nullptr))
473  multi_stream_max_async_queries = static_cast<int>(val);
474 
475  // keys that apply to Server
476  lookupkv(model, "Server", "target_latency_percentile", nullptr,
477  &server_target_latency_percentile, 0.01);
478  lookupkv(model, "Server", "target_latency", &server_target_latency_ns,
479  nullptr, 1000 * 1000);
480  lookupkv(model, "Server", "target_qps", nullptr, &server_target_qps);
481  if (lookupkv(model, "Server", "coalesce_queries", &val, nullptr))
482  server_coalesce_queries = (val == 0) ? false : true;
483  if (lookupkv(model, "Server", "max_async_queries", &val, nullptr))
484  server_max_async_queries = int(val);
485 
486  // keys that apply to Offline
487  lookupkv(model, "Offline", "target_qps", 0, &offline_expected_qps);
488 
489  return 0;
490 }
