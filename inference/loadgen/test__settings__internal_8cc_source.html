21 TestSettingsInternal::TestSettingsInternal(
22  const TestSettings &requested_settings)
23  : requested(requested_settings),
24  scenario(requested.scenario),
25  mode(requested.mode),
26  samples_per_query(1),
27  target_qps(1),
28  max_async_queries(-1),
29  target_duration(std::chrono::milliseconds(requested.min_duration_ms)),
30  min_duration(std::chrono::milliseconds(requested.min_duration_ms)),
31  max_duration(std::chrono::milliseconds(requested.max_duration_ms)),
32  min_query_count(requested.min_query_count),
33  max_query_count(requested.max_query_count),
34  min_sample_count(0),
35  qsl_rng_seed(requested.qsl_rng_seed),
36  sample_index_rng_seed(requested.sample_index_rng_seed),
37  schedule_rng_seed(requested.schedule_rng_seed) {
38  // Target QPS, target latency, and max_async_queries.
39  switch (requested.scenario) {
40  case TestScenario::SingleStream:
41  target_qps = static_cast<double>(std::nano::den) /
42  requested.single_stream_expected_latency_ns;
43  max_async_queries = 1;
44  break;
45  case TestScenario::MultiStream:
46  case TestScenario::MultiStreamFree:
47  target_qps = requested.multi_stream_target_qps;
48  target_latency =
49  std::chrono::nanoseconds(requested.multi_stream_target_latency_ns);
50  max_async_queries = requested.multi_stream_max_async_queries;
51  break;
52  case TestScenario::Server:
53  if (requested.server_target_qps >= 0.0) {
54  target_qps = requested.server_target_qps;
55  } else {
56  LogDetail([server_target_qps = requested.server_target_qps,
57  target_qps = target_qps](AsyncDetail &detail) {
58  detail.Error("Invalid value for server_target_qps requested.",
59  "requested", server_target_qps, "using", target_qps);
60  });
61  }
62  target_latency =
63  std::chrono::nanoseconds(requested.server_target_latency_ns);
64  max_async_queries =
65  std::numeric_limits<decltype(max_async_queries)>::max();
66  break;
67  case TestScenario::Offline:
68  if (requested.offline_expected_qps >= 0.0) {
69  target_qps = requested.offline_expected_qps;
70  } else {
71  LogDetail([offline_expected_qps = requested.offline_expected_qps,
72  target_qps = target_qps](AsyncDetail &detail) {
73  detail.Error("Invalid value for offline_expected_qps requested.",
74  "requested", offline_expected_qps, "using", target_qps);
75  });
76  }
77  max_async_queries =
78  std::numeric_limits<decltype(max_async_queries)>::max();
79  break;
80  }
81 
82  // Samples per query.
83  if (requested.scenario == TestScenario::MultiStream ||
84  requested.scenario == TestScenario::MultiStreamFree) {
85  samples_per_query = requested.multi_stream_samples_per_query;
86  }
87 
88  // In the offline scenario, coalesce all queries into a single query.
89  if (requested.scenario == TestScenario::Offline) {
90  // TODO: Should the spec require a max duration for large query counts?
91  // kSlack is used to make sure we generate enough samples for the SUT
92  // to take longer than than the minimum test duration required by the
93  // MLPerf spec.
94  constexpr double kSlack = 1.1;
95  int target_sample_count =
96  kSlack * DurationToSeconds(target_duration) * target_qps;
97  samples_per_query = std::max<int>(min_query_count, target_sample_count);
98  min_query_count = 1;
99  target_duration = std::chrono::milliseconds(0);
100  }
101 
102  min_sample_count = min_query_count * samples_per_query;
103 }

105 std::string ToString(TestScenario scenario) {
106  switch (scenario) {
107  case TestScenario::SingleStream:
108  return "Single Stream";
109  case TestScenario::MultiStream:
110  return "Multi Stream";
111  case TestScenario::MultiStreamFree:
112  return "Multi Stream Free";
113  case TestScenario::Server:
114  return "Server";
115  case TestScenario::Offline:
116  return "Offline";
117  }
118  assert(false);
119  return "InvalidScenario";
120 }

137 void LogRequestedTestSettings(const TestSettings &s) {
138  LogDetail([s](AsyncDetail &detail) {
139  detail("");
140  detail("Requested Settings:");

142  detail("Test mode : " + ToString(s.mode));
143 
144  // Scenario-specific
145  switch (s.scenario) {
146  case TestScenario::SingleStream:
147  detail("single_stream_expected_latency_ns : ",
148  s.single_stream_expected_latency_ns);
149  break;
150  case TestScenario::MultiStream:
151  case TestScenario::MultiStreamFree:
152  detail("multi_stream_target_qps : ", s.multi_stream_target_qps);
153  detail("multi_stream_target_latency_ns : ",
154  s.multi_stream_target_latency_ns);
155  detail("multi_stream_samples_per_query : ",
156  s.multi_stream_samples_per_query);
157  detail("multi_stream_max_async_queries : ",
158  s.multi_stream_max_async_queries);
159  break;
160  case TestScenario::Server:
161  detail("server_target_qps : ", s.server_target_qps);
162  detail("server_target_latency_ns : ", s.server_target_latency_ns);
163  detail("server_coalesce_queries : ", s.server_coalesce_queries);
164  break;
165  case TestScenario::Offline:
166  detail("offline_expected_qps : ", s.offline_expected_qps);
167  break;
168  }
169 
170  // Overrides
171  detail("min_duration_ms : ", s.min_duration_ms);
172  detail("max_duration_ms : ", s.max_duration_ms);
173  detail("min_query_count : ", s.min_query_count);
174  detail("max_query_count : ", s.max_query_count);
175  detail("qsl_rng_seed : ", s.qsl_rng_seed);
176  detail("sample_index_rng_seed : ", s.sample_index_rng_seed);

179  detail("");
