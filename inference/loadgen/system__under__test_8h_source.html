36 class SystemUnderTest {
37  public:
38  virtual ~SystemUnderTest() {}

40  /// \brief A human-readable string for logging purposes.
41  virtual const std::string& Name() const = 0;
42 
43  /// \brief Lets the loadgen issue N samples to the SUT.
44  /// \details The SUT may either a) return immediately and signal completion
45  /// at a later time on another thread or b) it may block and signal
46  /// completion on the current stack. The load generator will handle both
47  /// cases properly.
48  /// Note: The data for neighboring samples may or may not be contiguous
49  /// depending on the scenario.
50  virtual void IssueQuery(const std::vector<QuerySample>& samples) = 0;
51 
52  /// \brief Called immediately after the last call to IssueQuery
53  /// in a series is made.
54  /// \details This doesn't necessarily signify the end of the
55  /// test since there may be multiple series involved during a test; for
56  /// example in accuracy mode.
57  /// Clients can use this to flush any deferred queries immediately, rather
58  /// than waiting for some timeout.
59  /// This is especially useful in the server scenario.
60  virtual void FlushQueries() = 0;
61 
62  /// \brief Reports the raw latency results to the SUT of each sample issued as
63  /// recorded by the load generator. Units are nanoseconds.
64  virtual void ReportLatencyResults(
65  const std::vector<QuerySampleLatency>& latencies_ns) = 0;
66 };
