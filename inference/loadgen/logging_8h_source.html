531 /// \brief ScopedTracer is an RAII object that traces the start and end
532 /// of its lifetime.
533 template <typename LambdaT>
534 class ScopedTracer {
535  public:
536  ScopedTracer(LambdaT&& lambda)
537  : start_(PerfClock::now()), lambda_(std::forward<LambdaT>(lambda)) {}
538 
539  ~ScopedTracer() {
540  Log([start = start_, lambda = std::move(lambda_),
541  end = PerfClock::now()](AsyncLog& log) {
542  log.SetScopedTraceTimes(start, end);
543  AsyncTrace async_trace(log);
544  lambda(async_trace);
545  });
546  }
547 
548  private:
549  PerfClock::time_point start_;
550  LambdaT lambda_;
551 };

553 /// \brief Helper that creates a ScopeTracer with automatic type deduction.
554 /// \details Helps with automatic template type deduction, which has been
555 /// supported for functions for a long time.
556 /// C++17 will support deduction for classes, which will neutralize the utility
557 /// of a helper function like this.
558 /// \todo Determine which traces to keep for submission purposes.
559 template <typename LambdaT>
560 auto MakeScopedTracer(LambdaT&& lambda) -> ScopedTracer<LambdaT> {
561  return ScopedTracer<LambdaT>(std::forward<LambdaT>(lambda));

564 template <typename... Args>
565 void AsyncLog::LogSummary(const std::string& message, const Args... args) {
566  auto tracer = MakeScopedTracer([message](AsyncTrace& trace) {
567  std::string sanitized_message = message;
568  std::replace(sanitized_message.begin(), sanitized_message.end(), '"', '\'');
569  std::replace(sanitized_message.begin(), sanitized_message.end(), '\n', ';');
570  trace("LogSummary", "message", "\"" + sanitized_message + "\"");
571  });
572  std::unique_lock<std::mutex> lock(log_mutex_);
573  *summary_out_ << message;
574  LogArgs(summary_out_, args...);
575  *summary_out_ << "\n";
576 
577  if (copy_summary_to_stdout_) {
578  std::cout << message;
579  LogArgs(&std::cout, args...);
580  std::cout << "\n";
581  }
582 }

584 template <typename... Args>
585 void AsyncLog::LogDetail(const std::string& message, const Args... args) {
586  auto tracer = MakeScopedTracer([message](AsyncTrace& trace) {
587  std::string sanitized_message = message;
588  std::replace(sanitized_message.begin(), sanitized_message.end(), '"', '\'');
589  std::replace(sanitized_message.begin(), sanitized_message.end(), '\n', ';');
590  trace("LogDetail", "message", "\"" + sanitized_message + "\"");
591  });
592  std::unique_lock<std::mutex> lock(log_mutex_);
593  std::vector<std::ostream*> detail_streams{detail_out_, &std::cout};
594  if (!copy_detail_to_stdout_) {
595  detail_streams.pop_back();
596  }
597  for (auto os : detail_streams) {
598  *os << "\"pid\": " << current_pid_ << ", "
599  << "\"tid\": " << current_tid_ << ", "
600  << "\"ts\": " << (log_detail_time_ - log_origin_).count() << "ns : ";
601  if (error_flagged_) {
602  *os << "ERROR : ";
603  } else if (warning_flagged_) {
604  *os << "WARNING : ";
605  }
606  *os << message;
607  LogArgs(os, args...);
608  *os << "\n";
609  if (error_flagged_) {
610  os->flush();
611  }
612  }
613  error_flagged_ = false;
614  warning_flagged_ = false;
615 }
