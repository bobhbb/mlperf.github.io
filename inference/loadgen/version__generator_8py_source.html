27 # Creates a C++ raw string literal using a delimiter that is very
28 # unlikely to show up in a git stats.
29 def make_raw_string(str) :
30  delimeter = "LGVG_RSLD"
31  return "R\"" + delimeter + "(" + str + ")" + delimeter + "\""

33 def func_def(name, string):
34  return ("const std::string& Loadgen" + name + "() {\n" +
35  " static const std::string str = " + string + ";\n" +
36  " return str;\n" +
37  "}\n\n")

40 # For clients that build the loadgen from the git respository without
41 # any modifications.
42 def generate_loadgen_version_definitions_git(ofile, git_command):
43  git_rev = os.popen(git_command + "rev-parse --short=10 HEAD").read()
44  git_commit_date = os.popen(git_command + "log --format=\"%cI\" -n 1").read()
45  git_status = os.popen(git_command + "status -s -uno").read()
46  git_log = os.popen(
47  git_command + "log --pretty=oneline -n 16 --no-decorate").read()
48  ofile.write(func_def("GitRevision", "\"" + git_rev[0:-1] + "\""))
49  ofile.write(func_def("GitCommitDate", "\"" + git_commit_date[0:-1] + "\""))
50  ofile.write(func_def("GitStatus", make_raw_string(git_status[0:-1])))
51  ofile.write(func_def("GitLog", make_raw_string(git_log[0:-1])))

64 # Always log the sha1 of the loadgen files, regardless of whether we are
65 # in the original git repository or not.
66 def generate_loadgen_version_definitions_sha1(ofile, loadgen_root):
67  """Writes definition for Sha1OfFiles."""
68  sha1s = ""
69  loadgen_files = (
70  ["/bindings/" + s for s in os.listdir(loadgen_root + "/bindings")] +
71  ["/demos/" + s for s in os.listdir(loadgen_root + "/demos")] +
72  ["/" + s for s in os.listdir(loadgen_root)])
73  for fn in sorted(loadgen_files):
74  full_fn = loadgen_root + fn
75  if not os.path.isfile(full_fn):
76  continue
77  file_data = open(full_fn, "rb").read()
78  sha1s += hashlib.sha1(file_data).hexdigest() + " " + fn + "\n"
79 
80  ofile.write(func_def("Sha1OfFiles", make_raw_string(sha1s[0:-1])))

83 # Outputs version function definitions to cc_filename.
84 # Includes SHA1's of the relevant dirs in the loadgen_root directory.
85 def generate_loadgen_version_definitions(cc_filename, loadgen_root):
86  """Generates the C++ source file with the loadgen version info."""
87  try:
88  os.makedirs(os.path.dirname(cc_filename))
89  except OSError as exc:
90  if exc.errno != errno.EEXIST:
91  raise
92  ofile = open(cc_filename, "w")
93  ofile.write("// DO NOT EDIT: Autogenerated by version_generator.py.\n\n")
94  ofile.write("#include <string>\n\n")
95  ofile.write("namespace mlperf {\n\n")
96  ofile.write(func_def("Version", "\".5a1\""))
97 
98  date_time_now_local = datetime.datetime.now().isoformat()
99  date_time_now_utc = datetime.datetime.utcnow().isoformat()
100  ofile.write(func_def("BuildDateLocal", "\"" + date_time_now_local + "\""))
101  ofile.write(func_def("BuildDateUtc", "\"" + date_time_now_utc + "\""))
102 
103  git_dir = "--git-dir=\"" + loadgen_root + "/../.git\" "
104  git_work_tree = "--work-tree=\"" + loadgen_root + "/..\" "
105  git_command = "git " + git_dir + git_work_tree
106  git_status = os.popen(git_command + "status")
107  git_status.read()
108  is_git_repo = git_status.close() is None
109  if is_git_repo:
110  generate_loadgen_version_definitions_git(ofile, git_command)
111  else:
112  generate_loadgen_verstion_definitions_git_stubs(ofile)
113  generate_loadgen_version_definitions_sha1(ofile, loadgen_root)
114 
115  ofile.write("} // namespace mlperf\n")
116  ofile.close()
