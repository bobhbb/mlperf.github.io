81 /// \brief Maintains data and timing info for a query and all its samples.
82 class QueryMetadata {
83  public:
84  QueryMetadata(const std::vector<QuerySampleIndex>& query_sample_indices,
85  std::chrono::nanoseconds scheduled_delta,
86  ResponseDelegate* response_delegate, SequenceGen* sequence_gen)
87  : scheduled_delta(scheduled_delta),
88  response_delegate(response_delegate),
89  sequence_id(sequence_gen->NextQueryId()),
90  wait_count_(query_sample_indices.size()) {
91  samples_.reserve(query_sample_indices.size());
92  for (QuerySampleIndex qsi : query_sample_indices) {
93  samples_.push_back({this, sequence_gen->NextSampleId(), qsi});
94  }
95  query_to_send.reserve(query_sample_indices.size());
96  for (auto& s : samples_) {
97  query_to_send.push_back(
98  {reinterpret_cast<ResponseId>(&s), s.sample_index});
99  }
100  }
101 
102  QueryMetadata(QueryMetadata&& src)
103  : query_to_send(std::move(src.query_to_send)),
104  scheduled_delta(src.scheduled_delta),
105  response_delegate(src.response_delegate),
106  sequence_id(src.sequence_id),
107  wait_count_(src.samples_.size()),
108  samples_(std::move(src.samples_)) {
109  // The move constructor should only be called while generating a
110  // vector of QueryMetadata, before it's been used.
111  // Assert that wait_count_ is in its initial state.
112  assert(src.wait_count_.load() == samples_.size());
113  // Update the "parent" of each sample to be this query; the old query
114  // address will no longer be valid.
115  // TODO: Only set up the sample parenting once after all the queries have
116  // been created, rather than re-parenting on move here.
117  for (size_t i = 0; i < samples_.size(); i++) {
118  SampleMetadata* s = &samples_[i];
119  s->query_metadata = this;
120  query_to_send[i].id = reinterpret_cast<ResponseId>(s);
121  }
122  }
123 
124  void NotifyOneSampleCompleted(PerfClock::time_point timestamp) {
125  size_t old_count = wait_count_.fetch_sub(1, std::memory_order_relaxed);
126  if (old_count == 1) {
127  all_samples_done_time = timestamp;
128  all_samples_done_.set_value();
129  response_delegate->QueryComplete();
130  }
131  }
132 
133  void WaitForAllSamplesCompleted() { all_samples_done_.get_future().wait(); }
134 
135  PerfClock::time_point WaitForAllSamplesCompletedWithTimestamp() {
136  all_samples_done_.get_future().wait();
137  return all_samples_done_time;
138  }
139 
140  public:
141  std::vector<QuerySample> query_to_send;
142  const std::chrono::nanoseconds scheduled_delta;
143  ResponseDelegate* const response_delegate;
144  const uint64_t sequence_id;
145 
146  // Performance information.
147 
148  int scheduled_intervals = 0; // Number of intervals between queries, as
149  // actually scheduled during the run.
150  // For the multi-stream scenario only.
151  PerfClock::time_point scheduled_time;
152  PerfClock::time_point issued_start_time;
153  PerfClock::time_point all_samples_done_time;
154 
155  private:
156  std::atomic<size_t> wait_count_;
157  std::promise<void> all_samples_done_;
158  std::vector<SampleMetadata> samples_;
159 };

171 /// \brief ResponseDelegate implementation templated by scenario and mode.
172 template <TestScenario scenario, TestMode mode>
173 struct ResponseDelegateDetailed : public ResponseDelegate {
174  std::atomic<size_t> queries_completed{0};

176  void SampleComplete(SampleMetadata* sample, QuerySampleResponse* response,
177  PerfClock::time_point complete_begin_time) override {
178  // Using a raw pointer here should help us hit the std::function
179  // small buffer optimization code path when we aren't copying data.
180  // For some reason, using std::unique_ptr<std::vector> wasn't moving
181  // into the lambda; even with C++14.
182  std::vector<uint8_t>* sample_data_copy = nullptr;
183  if (mode == TestMode::AccuracyOnly) {
184  // TODO: Verify accuracy with the data copied here.
185  uint8_t* src_begin = reinterpret_cast<uint8_t*>(response->data);
186  uint8_t* src_end = src_begin + response->size;
187  sample_data_copy = new std::vector<uint8_t>(src_begin, src_end);
188  }
189  Log([sample, complete_begin_time, sample_data_copy](AsyncLog& log) {
190  QueryMetadata* query = sample->query_metadata;
191  DurationGeneratorNs sched{query->scheduled_time};
192 
193  if (scenario == TestScenario::Server) {
194  // Trace the server scenario as a stacked graph via counter events.
195  DurationGeneratorNs issued{query->issued_start_time};
196  log.TraceCounterEvent("Latency", query->scheduled_time, "issue_delay",
197  sched.delta(query->issued_start_time),
198  "issue_to_done",
199  issued.delta(complete_begin_time));
200  } else if (scenario != TestScenario::Offline) {
201  // Disable tracing of each sample in offline mode, where visualizing
202  // all samples overlapping isn't practical.
203  log.TraceSample("Sample", sample->sequence_id, query->scheduled_time,
204  complete_begin_time, "sample_seq", sample->sequence_id,
205  "query_seq", query->sequence_id, "sample_idx",
206  sample->sample_index, "issue_start_ns",
207  sched.delta(query->issued_start_time), "complete_ns",
208  sched.delta(complete_begin_time));
209  }
210 
211  if (sample_data_copy) {
212  log.LogAccuracy(sample->sequence_id, sample->sample_index,
213  LogBinaryAsHexString{sample_data_copy});
214  delete sample_data_copy;
215  }
216 
217  // Record the latency at the end, since it will unblock the issuing
218  // thread and potentially destroy the metadata being used above.
219  QuerySampleLatency latency = sched.delta(complete_begin_time);
220  log.RecordLatency(sample->sequence_id, latency);
221  });
222  }
223 
224  void QueryComplete() override {
225  // We only need to track outstanding queries in the server scenario to
226  // detect when the SUT has fallen too far behind.
227  if (scenario == TestScenario::Server) {
228  queries_completed.fetch_add(1, std::memory_order_relaxed);
229  }
230  }

233 /// \brief Selects the query timestamps for all scenarios except Server.
234 template <TestScenario scenario>
235 auto ScheduleDistribution(double qps) {
236  return [period = std::chrono::duration_cast<std::chrono::nanoseconds>(
237  std::chrono::duration<double>(1.0 / qps))](auto& /*gen*/) {
238  return period;
239  };
240 }
241 
242 /// \brief Selects the query timestamps for the Server scenario.
243 template <>
244 auto ScheduleDistribution<TestScenario::Server>(double qps) {
245  // Poisson arrival process corresponds to exponentially distributed
246  // interarrival times.
247  return [dist = std::exponential_distribution<>(qps)](auto& gen) mutable {
248  return std::chrono::duration_cast<std::chrono::nanoseconds>(
249  std::chrono::duration<double>(dist(gen)));
250  };
251 }
252 
253 /// \brief Selects samples for the accuracy mode.
254 template <TestMode mode>
255 auto SampleDistribution(size_t sample_count, size_t stride, std::mt19937* rng) {
256  std::vector<size_t> indices;
257  for (size_t i = 0; i < sample_count; i += stride) {
258  indices.push_back(i);
259  }
260  std::shuffle(indices.begin(), indices.end(), *rng);
261  return
262  [ indices = std::move(indices), i = size_t(0) ](auto& /*gen*/) mutable {
263  return indices.at(i++);
264  };
265 }
266 
267 /// \brief Selects samples for the performance mode.
268 /// \todo Contiguous samples for the offline scenario.
269 template <>
270 auto SampleDistribution<TestMode::PerformanceOnly>(size_t sample_count,
271  size_t /*stride*/,
272  std::mt19937* /*rng*/) {
273  return [dist = std::uniform_int_distribution<>(0, sample_count - 1)](
274  auto& gen) mutable {
275  return dist(gen);
276  };
277 }

279 /// \brief Generates queries for the requested settings, templated by
280 /// scenario and mode.
281 /// \todo Make GenerateQueries faster.
282 /// QueryMetadata is expensive to move; either reserve queries in advance
283 /// so the queries vector doesn't need to grow. And/or parent samples to their
284 /// queries only after all queries have been generated.
285 /// \todo For the server scenario only, scale the query timeline at the end so
286 /// the QPS as scheduled is equal to the QPS as requested.
287 template <TestScenario scenario, TestMode mode>
288 std::vector<QueryMetadata> GenerateQueries(
289  const TestSettingsInternal& settings,
290  const LoadableSampleSet& loaded_sample_set, SequenceGen* sequence_gen,
291  ResponseDelegate* response_delegate) {
292  auto tracer =
293  MakeScopedTracer([](AsyncTrace& trace) { trace("GenerateQueries"); });
294 
295  auto& loaded_samples = loaded_sample_set.set;
296 
297  // Generate 2x more samples than we think we'll need given the expected
298  // QPS. We should exit before issuing all queries.
299  std::chrono::microseconds k2xTargetDuration = 2 * settings.target_duration;
300  size_t min_queries = settings.min_query_count;
301 
302  // We should not exit early in accuracy mode.
303  if (mode == TestMode::AccuracyOnly) {
304  k2xTargetDuration = std::chrono::microseconds(0);
305  // Integer truncation here is intentional.
306  // For MultiStream, loaded samples is properly padded.
307  // For Offline, we create a 'remainder' query at the end of this function.
308  min_queries = loaded_samples.size() / settings.samples_per_query;
309  }
310 
311  std::vector<QueryMetadata> queries;
312 
313  // Using the std::mt19937 pseudo-random number generator ensures a modicum of
314  // cross platform reproducibility for trace generation.
315  std::mt19937 sample_rng(settings.sample_index_rng_seed);
316  std::mt19937 schedule_rng(settings.schedule_rng_seed);
317 
318  constexpr bool kIsMultiStream = scenario == TestScenario::MultiStream ||
319  scenario == TestScenario::MultiStreamFree;
320  const size_t sample_stride = kIsMultiStream ? settings.samples_per_query : 1;
321 
322  auto sample_distribution = SampleDistribution<mode>(
323  loaded_sample_set.sample_distribution_end, sample_stride, &sample_rng);
324  auto schedule_distribution =
325  ScheduleDistribution<scenario>(settings.target_qps);
326 
327  size_t samples_per_query = settings.samples_per_query;
328  if (mode == TestMode::AccuracyOnly && scenario == TestScenario::Offline) {
329  samples_per_query = loaded_sample_set.sample_distribution_end;
330  }
331 
332  std::vector<QuerySampleIndex> samples(samples_per_query);
333  std::chrono::nanoseconds timestamp(0);
334  while (timestamp <= k2xTargetDuration || queries.size() < min_queries) {
335  if (kIsMultiStream) {
336  QuerySampleIndex sample_i = sample_distribution(sample_rng);
337  for (auto& s : samples) {
338  // Select contiguous samples in the MultiStream scenario.
339  // This will not overflow, since GenerateLoadableSets adds padding at
340  // the end of the loadable sets in the MultiStream scenario.
341  // The padding allows the starting samples to be the same for each
342  // query as the value of samples_per_query increases.
343  s = loaded_samples[sample_i++];
344  }
345  } else if(mode == TestMode::PerformanceOnly && scenario == TestScenario::Offline ){
346  // For the Offline + Performance scenario, we also want to support
347  // contiguous samples. In this scenario the query can be much larger than what
348  // fits into memory. We simply repeat loaded_samples N times, plus a remainder
349  // to ensure we fill up samples.
350  // Note that this eliminates randomization.
351  size_t num_loaded_samples = loaded_samples.size();
352  size_t num_full_repeats = samples_per_query/num_loaded_samples;
353  int remainder = samples_per_query % (num_loaded_samples);
354 
355  for (int i = 0; i < num_full_repeats; ++i){
356  std::copy (loaded_samples.begin(), loaded_samples.end(), samples.begin() + i * num_loaded_samples);
357  }
358 
359  std::copy(loaded_samples.begin(), loaded_samples.begin() + remainder, samples.begin() + num_full_repeats * num_loaded_samples);
360 
361  } else {
362  for (auto& s : samples) {
363  s = loaded_samples[sample_distribution(sample_rng)];
364  }
365  }
366  queries.emplace_back(samples, timestamp, response_delegate, sequence_gen);
367  timestamp += schedule_distribution(schedule_rng);
368  }
369 
370  // See if we need to create a "remainder" query for offline+accuracy to
371  // ensure we issue all samples in loaded_samples. Offline doesn't pad
372  // loaded_samples like MultiStream does.
373  if (scenario == TestScenario::Offline && mode == TestMode::AccuracyOnly) {
374  size_t remaining_samples = loaded_samples.size() % samples_per_query;
375  if (remaining_samples != 0) {
376  samples.resize(remaining_samples);
377  for (auto& s : samples) {
378  s = loaded_samples[sample_distribution(sample_rng)];
379  }
380  queries.emplace_back(samples, timestamp, response_delegate, sequence_gen);
381  }
382  }
383 
384  LogDetail([
385  count = queries.size(), spq = settings.samples_per_query,
386  duration = timestamp.count()
387  ](AsyncDetail & detail) {
388  detail("GeneratedQueries: ", "queries", count, "samples per query", spq,
389  "duration", duration);
390  });
391 
392  return queries;
393 }

424 /// \brief Schedules queries for issuance in the multi stream scenario.
425 template <>
426 struct QueryScheduler<TestScenario::MultiStream> {
427  QueryScheduler(const TestSettingsInternal& settings,
428  const PerfClock::time_point start)
429  : qps(settings.target_qps),
430  max_async_queries(settings.max_async_queries),
431  start_time(start) {}
432 
433  PerfClock::time_point Wait(QueryMetadata* next_query) {
434  {
435  prev_queries.push(next_query);
436  auto tracer =
437  MakeScopedTracer([](AsyncTrace& trace) { trace("Waiting"); });
438  if (prev_queries.size() > max_async_queries) {
439  prev_queries.front()->WaitForAllSamplesCompleted();
440  prev_queries.pop();
441  }
442  }
443 
444  {
445  auto tracer =
446  MakeScopedTracer([](AsyncTrace& trace) { trace("Scheduling"); });
447  // TODO(brianderson): Skip ticks based on the query complete time,
448  // before the query snchronization + notification thread hop,
449  // rather than after.
450  PerfClock::time_point now = PerfClock::now();
451  auto i_period_old = i_period;
452  PerfClock::time_point tick_time;
453  do {
454  i_period++;
455  tick_time =
456  start_time + SecondsToDuration<PerfClock::duration>(i_period / qps);
457  Log([tick_time](AsyncLog& log) {
458  log.TraceAsyncInstant("QueryInterval", 0, tick_time);
459  });
460  } while (tick_time < now);
461  next_query->scheduled_intervals = i_period - i_period_old;
462  next_query->scheduled_time = tick_time;
463  std::this_thread::sleep_until(tick_time);
464  }
465 
466  auto now = PerfClock::now();
467  next_query->issued_start_time = now;
468  return now;
469  }
470 
471  size_t i_period = 0;
472  double qps;
473  const size_t max_async_queries;
474  PerfClock::time_point start_time;
475  std::queue<QueryMetadata*> prev_queries;
476 };

564 /// \brief Issues a series of pre-generated queries.
565 // TODO: Templates for scenario and mode are overused, given the loadgen
566 // no longer generates queries on the fly. Should we reduce the
567 // use of templates?
568 template <TestScenario scenario, TestMode mode>
569 PerformanceResult IssueQueries(SystemUnderTest* sut,
570  const TestSettingsInternal& settings,
571  const LoadableSampleSet& loaded_sample_set,
572  SequenceGen* sequence_gen) {
573  GlobalLogger().RestartLatencyRecording(sequence_gen->CurrentSampleId());
574  ResponseDelegateDetailed<scenario, mode> response_logger;
575 
576  std::vector<QueryMetadata> queries = GenerateQueries<scenario, mode>(
577  settings, loaded_sample_set, sequence_gen, &response_logger);
578 
579  size_t queries_issued = 0;
580  // TODO: Replace the constant 5 below with a TestSetting.
581  const double query_seconds_outstanding_threshold =
582  5 *
583  std::chrono::duration_cast<std::chrono::duration<double>>(
584  settings.target_latency)
585  .count();
586  const size_t max_queries_outstanding =
587  settings.target_qps * query_seconds_outstanding_threshold;
588 
589  const PerfClock::time_point start = PerfClock::now();
590  PerfClock::time_point last_now = start;
591  QueryScheduler<scenario> query_scheduler(settings, start);
592 
593  size_t expected_latencies = 0;
594  for (auto& query : queries) {
595  auto tracer1 =
596  MakeScopedTracer([](AsyncTrace& trace) { trace("SampleLoop"); });
597  last_now = query_scheduler.Wait(&query);
598 
599  // Issue the query to the SUT.
600  {
601  auto tracer3 =
602  MakeScopedTracer([](AsyncTrace& trace) { trace("IssueQuery"); });
603  sut->IssueQuery(query.query_to_send);
604  }
605 
606  expected_latencies += query.query_to_send.size();
607  queries_issued++;
608 
609  if (mode == TestMode::AccuracyOnly) {
610  // TODO: Rate limit in accuracy mode so accuracy mode works even
611  // if the expected/target performance is way off.
612  continue;
613  }
614 
615  auto duration = (last_now - start);
616  if (queries_issued >= settings.min_query_count &&
617  duration > settings.min_duration) {
618  LogDetail([](AsyncDetail& detail) {
619  detail("Ending naturally: Minimum query count and test duration met.");
620  });
621  break;
622  }
623  if (settings.max_query_count != 0 &&
624  queries_issued >= settings.max_query_count) {
625  LogDetail([queries_issued](AsyncDetail& detail) {
626  detail.Error("Ending early: Max query count reached.", "query_count",
627  queries_issued);
628  });
629  break;
630  }
631  if (settings.max_duration.count() != 0 &&
632  duration > settings.max_duration) {
633  LogDetail([duration](AsyncDetail& detail) {
634  detail.Error("Ending early: Max test duration reached.", "duration_ns",
635  duration.count());
636  });
637  break;
638  }
639  if (scenario == TestScenario::Server) {
640  size_t queries_outstanding =
641  queries_issued -
642  response_logger.queries_completed.load(std::memory_order_relaxed);
643  if (queries_outstanding > max_queries_outstanding) {
644  LogDetail([queries_issued, queries_outstanding](AsyncDetail& detail) {
645  detail.Error("Ending early: Too many outstanding queries.", "issued",
646  queries_issued, "outstanding", queries_outstanding);
647  });
648  break;
649  }
650  }
651  }
652 
653  // Let the SUT know it should not expect any more queries.
654  sut->FlushQueries();
655 
656  // The offline scenario always only has a single query, so this check
657  // doesn't apply.
658  if (scenario != TestScenario::Offline && mode == TestMode::PerformanceOnly &&
659  queries_issued >= queries.size()) {
660  LogDetail([](AsyncDetail& detail) {
661  detail.Error(
662  "Ending early: Ran out of generated queries to issue before the "
663  "minimum query count and test duration were reached.");
664  detail(
665  "Please update the relevant expected latency or target qps in the "
666  "TestSettings so they are more accurate.");
667  });
668  }
669 
670  // Wait for tail queries to complete and collect all the latencies.
671  // We have to keep the synchronization primitives alive until the SUT
672  // is done with them.
673  auto& final_query = queries[queries_issued - 1];
674  std::vector<QuerySampleLatency> latencies(
675  GlobalLogger().GetLatenciesBlocking(expected_latencies));
676 
677  // Log contention counters after every test as a sanity check.
678  GlobalLogger().LogContentionCounters();
679 
680  double max_latency =
681  QuerySampleLatencyToSeconds(GlobalLogger().GetMaxLatencySoFar());
682  double final_query_scheduled_time =
683  DurationToSeconds(final_query.scheduled_delta);
684  double final_query_issued_time =
685  DurationToSeconds(final_query.issued_start_time - start);
686  double final_query_all_samples_done_time =
687  DurationToSeconds(final_query.all_samples_done_time - start);
688  return PerformanceResult{std::move(latencies),
689  queries_issued,
690  max_latency,
691  final_query_scheduled_time,
692  final_query_issued_time,
693  final_query_all_samples_done_time};
694 }

730 void PerformanceSummary::ProcessLatencies() {
731  if (pr.latencies.empty()) {
732  return;
733  }
734 
735  sample_count = pr.latencies.size();
736 
737  QuerySampleLatency accumulated_latency = 0;
738  for (auto latency : pr.latencies) {
739  accumulated_latency += latency;
740  }
741  latency_mean = accumulated_latency / sample_count;
742 
743  std::sort(pr.latencies.begin(), pr.latencies.end());
744 
745  latency_target.value = pr.latencies[sample_count * latency_target.percentile];
746  latency_min = pr.latencies.front();
747  latency_max = pr.latencies.back();
748  for (auto& lp : latency_percentiles) {
749  assert(lp.percentile >= 0.0);
750  assert(lp.percentile < 1.0);
751  lp.value = pr.latencies[sample_count * lp.percentile];
752  }
753 
754  // Clear latencies since we are done processing them.
755  pr.latencies = std::vector<QuerySampleLatency>();
756 }

837 void PerformanceSummary::Log(AsyncSummary& summary) {
838  ProcessLatencies();
839 
840  summary(
841  "================================================\n"
842  "MLPerf Results Summary\n"
843  "================================================");
844  summary("SUT name : ", sut_name);
845  summary("Scenario : ", ToString(settings.scenario));
846  summary("Mode : ", ToString(settings.mode));
847 
848  switch (settings.scenario) {
849  case TestScenario::SingleStream: {
850  summary("90th percentile latency (ns) : ", latency_target.value);
851  break;
852  }
853  case TestScenario::MultiStream: {
854  summary("Samples per query : ", settings.samples_per_query);
855  break;
856  }
857  case TestScenario::MultiStreamFree: {
858  double samples_per_second = pr.queries_issued *
859  settings.samples_per_query /
860  pr.final_query_all_samples_done_time;
861  summary("Samples per second : ", samples_per_second);
862  break;
863  }
864  case TestScenario::Server: {
865  // Subtract 1 from sample count since the start of the final sample
866  // represents the open end of the time range: i.e. [begin, end).
867  // This makes sense since:
868  // a) QPS doesn't apply if there's only one sample; it's pure latency.
869  // b) If you have precisely 1k QPS, there will be a sample exactly on
870  // the 1 second time point; but that would be the 1001th sample in
871  // the stream. Given the first 1001 queries, the QPS is
872  // 1000 queries / 1 second.
873  double qps_as_scheduled =
874  (sample_count - 1) / pr.final_query_scheduled_time;
875  summary("Scheduled samples per second : ",
876  DoubleToString(qps_as_scheduled));
877  break;
878  }
879  case TestScenario::Offline: {
880  double samples_per_second = sample_count / pr.max_latency;
881  summary("Samples per second: ", samples_per_second);
882  break;
883  }
884  }
885 
886  std::string min_duration_recommendation;
887  std::string perf_constraints_recommendation;
888 
889  bool min_duration_met = MinDurationMet(&min_duration_recommendation);
890  bool min_queries_met = MinQueriesMet() && MinSamplesMet();
891  bool perf_constraints_met =
892  PerfConstraintsMet(&perf_constraints_recommendation);
893  bool all_constraints_met =
894  min_duration_met && min_queries_met && perf_constraints_met;
895  summary("Result is : ", all_constraints_met ? "VALID" : "INVALID");
896  if (HasPerfConstraints()) {
897  summary(" Performance constraints satisfied : ",
898  perf_constraints_met ? "Yes" : "NO");
899  }
900  summary(" Min duration satisfied : ", min_duration_met ? "Yes" : "NO");
901  summary(" Min queries satisfied : ", min_queries_met ? "Yes" : "NO");
902 
903  if (!all_constraints_met) {
904  summary("Recommendations:");
905  if (!perf_constraints_met) {
906  summary(" * " + perf_constraints_recommendation);
907  }
908  if (!min_duration_met) {
909  summary(" * " + min_duration_recommendation);
910  }
911  if (!min_queries_met) {
912  summary(
913  " * The test exited early, before enough queries were issued.\n"
914  " See the detailed log for why this may have occurred.");
915  }
916  }
917 
918  summary(
919  "\n"
920  "================================================\n"
921  "Additional Stats\n"
922  "================================================");
923 
924  if (settings.scenario == TestScenario::SingleStream) {
925  double qps_w_lg = (sample_count - 1) / pr.final_query_issued_time;
926  /// \todo qps_wo_log should use latency_mean
927  double qps_wo_lg = 1 / QuerySampleLatencyToSeconds(latency_min);
928  summary("QPS w/ loadgen overhead : " + DoubleToString(qps_w_lg));
929  summary("QPS w/o loadgen overhead : " + DoubleToString(qps_wo_lg));
930  summary("");
931  } else if (settings.scenario == TestScenario::Server) {
932  double qps_as_completed =
933  (sample_count - 1) / pr.final_query_all_samples_done_time;
934  summary("Completed samples per second : ",
935  DoubleToString(qps_as_completed));
936  summary("");
937  }
938 
939  summary("Min latency (ns) : ", latency_min);
940  summary("Max latency (ns) : ", latency_max);
941  summary("Mean latency (ns) : ", latency_mean);
942  for (auto& lp : latency_percentiles) {
943  summary(
944  DoubleToString(lp.percentile * 100) + " percentile latency (ns) : ",
945  lp.value);
946  }
947 
948  summary(
949  "\n"
950  "================================================\n"
951  "Test Parameters Used\n"
952  "================================================");
953  settings.LogSummary(summary);
954 }

1089 /// \brief Generates random sets of samples in the QSL that we can load into
1090 /// RAM at the same time.
1091 std::vector<LoadableSampleSet> GenerateLoadableSets(
1092  QuerySampleLibrary* qsl, const TestSettingsInternal& settings) {
1093  auto tracer = MakeScopedTracer(
1094  [](AsyncTrace& trace) { trace("GenerateLoadableSets"); });
1095 
1096  std::vector<LoadableSampleSet> result;
1097  std::mt19937 qsl_rng(settings.qsl_rng_seed);
1098 
1099  // Generate indices for all available samples in the QSL.
1100  const size_t qsl_total_count = qsl->TotalSampleCount();
1101  std::vector<QuerySampleIndex> samples(qsl_total_count);
1102  for (size_t i = 0; i < qsl_total_count; i++) {
1103  samples[i] = static_cast<QuerySampleIndex>(i);
1104  }
1105 
1106  // Randomize the order of the samples.
1107  std::shuffle(samples.begin(), samples.end(), qsl_rng);
1108 
1109  // Partition the samples into loadable sets.
1110  const size_t set_size = qsl->PerformanceSampleCount();
1111  const size_t set_padding =
1112  (settings.scenario == TestScenario::MultiStream ||
1113  settings.scenario == TestScenario::MultiStreamFree)
1114  ? settings.samples_per_query - 1
1115  : 0;
1116  std::vector<QuerySampleIndex> loadable_set;
1117  loadable_set.reserve(set_size + set_padding);
1118 
1119  for (auto s : samples) {
1120  loadable_set.push_back(s);
1121  if (loadable_set.size() == set_size) {
1122  result.push_back({std::move(loadable_set), set_size});
1123  loadable_set.clear();
1124  loadable_set.reserve(set_size + set_padding);
1125  }
1126  }
1127 
1128  if (!loadable_set.empty()) {
1129  // Copy the size since it will become invalid after the move.
1130  size_t loadable_set_size = loadable_set.size();
1131  result.push_back({std::move(loadable_set), loadable_set_size});
1132  }
1133 
1134  // Add padding for the multi stream scenario. Padding allows the
1135  // startings sample to be the same for all SUTs, independent of the value
1136  // of samples_per_query, while enabling samples in a query to be contiguous.
1137  for (auto& loadable_set : result) {
1138  auto& set = loadable_set.set;
1139  for (size_t i = 0; i < set_padding; i++) {
1140  // It's not clear in the spec if the STL deallocates the old container
1141  // before assigning, which would invalidate the source before the
1142  // assignment happens. Even though we should have reserved enough
1143  // elements above, copy the source first anyway since we are just moving
1144  // integers around.
1145  QuerySampleIndex p = set[i];
1146  set.push_back(p);
1147  }
1148  }
1149 
1150  return result;
1151 }
