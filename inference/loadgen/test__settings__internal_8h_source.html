92 template <TestScenario scenario>
93 TestSettingsInternal MidOfBoundaries(
94  const TestSettingsInternal &lower_bound_settings,
95  const TestSettingsInternal &upper_bound_settings) {
96  TestSettingsInternal mid_settings = lower_bound_settings;
97  if (scenario == TestScenario::MultiStream ||
98  scenario == TestScenario::MultiStreamFree) {
99  assert(lower_bound_settings.samples_per_query <
100  upper_bound_settings.samples_per_query);
101  mid_settings.samples_per_query = lower_bound_settings.samples_per_query +
102  (upper_bound_settings.samples_per_query -
103  lower_bound_settings.samples_per_query) /
104  2;
105  } else if (scenario == TestScenario::Server) {
106  assert(lower_bound_settings.target_qps < upper_bound_settings.target_qps);
107  mid_settings.target_qps =
108  lower_bound_settings.target_qps +
109  (upper_bound_settings.target_qps - lower_bound_settings.target_qps) / 2;
110  } else {
111  LogDetail([](AsyncDetail &detail) { detail(kNotSupportedMsg); });
112  }
113  return mid_settings;
114 }
115 
116 template <TestScenario scenario>
117 bool IsFinished(const TestSettingsInternal &lower_bound_settings,
118  const TestSettingsInternal &upper_bound_settings) {
119  if (scenario == TestScenario::MultiStream ||
120  scenario == TestScenario::MultiStreamFree) {
121  return lower_bound_settings.samples_per_query + 1 >=
122  upper_bound_settings.samples_per_query;
123  } else if (scenario == TestScenario::Server) {
124  uint8_t precision = lower_bound_settings.requested
125  .server_find_peak_qps_decimals_of_precision;
126  double l =
127  std::floor(lower_bound_settings.target_qps * std::pow(10, precision));
128  double u =
129  std::floor(upper_bound_settings.target_qps * std::pow(10, precision));
130  return l + 1 >= u;
131  } else {
132  LogDetail([](AsyncDetail &detail) { detail(kNotSupportedMsg); });
133  return true;
134  }
135 }

137 template <TestScenario scenario>
138 std::string ToStringPerformanceField(const TestSettingsInternal &settings) {
139  if (scenario == TestScenario::MultiStream ||
140  scenario == TestScenario::MultiStreamFree) {
141  return std::to_string(settings.samples_per_query);
142  } else if (scenario == TestScenario::Server) {
143  return std::to_string(settings.target_qps);
144  } else {
145  LogDetail([](AsyncDetail &detail) { detail(kNotSupportedMsg); });
146  return ToString(settings.scenario);
147  }
148 }

150 template <TestScenario scenario>
151 void WidenPerformanceField(TestSettingsInternal *settings) {
152  if (scenario == TestScenario::MultiStream ||
153  scenario == TestScenario::MultiStreamFree) {
154  settings->samples_per_query = settings->samples_per_query * 2;
155  } else if (scenario == TestScenario::Server) {
156  settings->target_qps =
157  settings->target_qps *
158  (1 + settings->requested.server_find_peak_qps_boundary_step_size);
159  } else {
160  LogDetail([](AsyncDetail &detail) { detail(kNotSupportedMsg); });
161  }
