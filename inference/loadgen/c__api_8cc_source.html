27 // Forwards SystemUnderTest calls to relevant callbacks.
28 class SystemUnderTestTrampoline : public SystemUnderTest {
29  public:
30  SystemUnderTestTrampoline(
31  ClientData client_data, std::string name, IssueQueryCallback issue_cb,
32  FlushQueriesCallback flush_queries_cb,
33  ReportLatencyResultsCallback report_latency_results_cb)
34  : client_data_(client_data),
35  name_(std::move(name)),
36  issue_cb_(issue_cb),
37  flush_queries_cb_(flush_queries_cb),
38  report_latency_results_cb_(report_latency_results_cb) {}
39  ~SystemUnderTestTrampoline() override = default;
40 
41  const std::string& Name() const override { return name_; }
42 
43  void IssueQuery(const std::vector<QuerySample>& samples) override {
44  (*issue_cb_)(client_data_, samples.data(), samples.size());
45  }
46 
47  void FlushQueries() override { (*flush_queries_cb_)(); }
48 
49  void ReportLatencyResults(
50  const std::vector<QuerySampleLatency>& latencies_ns) override {
51  (*report_latency_results_cb_)(client_data_, latencies_ns.data(),
52  latencies_ns.size());
53  }
54 
55  private:
56  ClientData client_data_;
57  std::string name_;
58  IssueQueryCallback issue_cb_;
59  FlushQueriesCallback flush_queries_cb_;
60  ReportLatencyResultsCallback report_latency_results_cb_;
