35 /// \brief Implements the client interfaces of the loadgen and
36 /// has some basic sanity checks that are enabled for all tests.
37 /// \details It also forwards calls to overrideable *Ext methods and implements
38 /// the TestProxy concept.
39 struct SystemUnderTestBasic : public mlperf::QuerySampleLibrary,
40  public mlperf::SystemUnderTest {
41  const std::string& Name() const override { return name_; }
42 
43  size_t TotalSampleCount() override { return total_sample_count_; }
44  size_t PerformanceSampleCount() override { return performance_sample_count_; }

46  void LoadSamplesToRam(
47  const std::vector<mlperf::QuerySampleIndex>& samples) override {
48  for (auto s : samples) {
49  samples_load_count_.at(s)++;
50  loaded_samples_.push_back(s);
51  }
52  LoadSamplesToRamExt(samples);
53  }
54  virtual void LoadSamplesToRamExt(
55  const std::vector<mlperf::QuerySampleIndex>& samples) {}
56 
57  void UnloadSamplesFromRam(
58  const std::vector<mlperf::QuerySampleIndex>& samples) override {
59  for (auto s : samples) {
60  FAIL_IF(loaded_samples_.front() != s) &&
61  FAIL_EXP(loaded_samples_.front()) && FAIL_EXP(s);
62  loaded_samples_.pop_front();
63  size_t prev_load_count = samples_load_count_.at(s)--;
64  FAIL_IF(prev_load_count == 0) && FAIL_EXP(prev_load_count);
65  }
66  UnloadSamplesFromRamExt(samples);
67  }
68  virtual void UnloadSamplesFromRamExt(
69  const std::vector<mlperf::QuerySampleIndex>& samples) {}
70 
71  void IssueQuery(const std::vector<mlperf::QuerySample>& samples) override {
72  std::vector<mlperf::QuerySampleResponse> responses;
73  query_sizes_.push_back(samples.size());
74  samples_between_flushes_.back() += samples.size();
75  responses.reserve(samples.size());
76  for (auto s : samples) {
77  FAIL_IF(samples_load_count_.at(s.index) == 0) &&
78  FAIL_MSG("Issued unloaded sample:") && FAIL_EXP(s.index);
79  samples_issue_count_.at(s.index)++;
80  issued_samples_.push_back(s.index);
81  responses.push_back({s.id, 0, 0});
82  }
83  mlperf::QuerySamplesComplete(responses.data(), responses.size());
84  IssueQueryExt(samples);
85  }
86  virtual void IssueQueryExt(const std::vector<mlperf::QuerySample>& samples) {}
87 
88  void FlushQueries() override {
89  samples_between_flushes_.push_back(0);
90  FlushQueriesExt();
91  }
92  virtual void FlushQueriesExt() {}
93 
94  void ReportLatencyResults(
95  const std::vector<mlperf::QuerySampleLatency>& latencies_ns) override {}
96 
97  virtual void RunTest() {
98  samples_load_count_.resize(total_sample_count_, 0);
99  samples_issue_count_.resize(total_sample_count_, 0);
100  samples_between_flushes_.resize(1, 0);
101  mlperf::StartTest(this, this, test_settings_, log_settings_);
102  }
103 
104  virtual void EndTest() {}
105 
106  protected:
107  mlperf::TestSettings test_settings_;
108  mlperf::LogSettings log_settings_;
109 
110  std::string name_{"BasicSUT"};
111  size_t total_sample_count_;
112  size_t performance_sample_count_;
113  std::vector<mlperf::QuerySampleIndex> issued_samples_;
114  std::deque<mlperf::QuerySampleIndex> loaded_samples_;
115  std::vector<size_t> samples_load_count_;
116  std::vector<size_t> samples_issue_count_;
117 
118  std::vector<size_t> query_sizes_;
119  std::vector<size_t> samples_between_flushes_;
120 };

144 /// \brief Verifies all samples from the QSL are included at least once
145 /// in accuracy mode.
146 /// \ingroup LoadgenTestsBasic
147 struct TestAccuracyIncludesAllSamples : public SystemUnderTestAccuracy {
148  void EndTest() override {
149  std::sort(issued_samples_.begin(), issued_samples_.end());
150 
151  FAIL_IF(issued_samples_.size() < total_sample_count_) &&
152  FAIL_EXP(issued_samples_.size()) && FAIL_EXP(total_sample_count_);
153  FAIL_IF(issued_samples_.front() != 0) && FAIL_EXP(issued_samples_.front());
154  FAIL_IF(issued_samples_.back() != total_sample_count_ - 1) &&
155  FAIL_EXP(issued_samples_.back()) && FAIL_EXP(total_sample_count_);
156 
157  mlperf::QuerySampleIndex prev = -1;
158  size_t discontinuities = 0;
159  size_t dupes = 0;
160  for (auto s : issued_samples_) {
161  if (s == prev) {
162  dupes++;
163  } else if (s - prev > 1) {
164  discontinuities++;
165  }
166  prev = s;
167  }
168 
169  FAIL_IF(discontinuities != 0) && FAIL_EXP(discontinuities);
170  if (test_settings_.scenario == mlperf::TestScenario::MultiStream ||
171  test_settings_.scenario == mlperf::TestScenario::MultiStreamFree) {
172  const size_t expected_sets =
173  total_sample_count_ / performance_sample_count_;
174  FAIL_IF(dupes >=
175  test_settings_.multi_stream_samples_per_query * expected_sets) &&
176  FAIL_EXP(dupes);
177  } else {
178  FAIL_IF(dupes != 0) && FAIL_EXP(dupes);
179  }
180  }
181 };

183 REGISTER_TEST_ALL_SCENARIOS(AccuracyIncludesAllSamples,

187 /// \brief Verifies offline + accuracy doesn't hang if the last set
188 /// in the accuracy series is smaller than others.
189 /// \ingroup LoadgenTestsBasic
190 struct TestOfflineRemainderAccuracySet : public SystemUnderTestAccuracy {
191  void SetUpTest() {
192  SystemUnderTestAccuracy::SetUpTest(4, 0, 7, mlperf::TestScenario::Offline);
193  }
194 
195  void EndTest() override {
196  auto& flush_samples = samples_between_flushes_;
197 
198  FAIL_IF(flush_samples.size() < 3) && FAIL_EXP(flush_samples.size()) &&
199  BAD_TEST_MSG("Test should generate multiple query sets.") && ABORT_TEST;
200 
201  // The last counter will be 0, since a test ends with a call to
202  // FlushQuery.
203  FAIL_IF(flush_samples.back() != 0) && FAIL_EXP(flush_samples.back()) &&
204  FAIL_MSG(
205  "Detected stray calls to IssueQuery after the last call to "
206  "FlushQuery.");
207  flush_samples.pop_back();
208 
209  // Verify the test ran with a smaller last accuracy set.
210  size_t first_size = flush_samples.front();
211  size_t last_size = flush_samples.back();
212  FAIL_IF(first_size <= last_size) && FAIL_EXP(first_size) &&
213  FAIL_EXP(last_size) && BAD_TEST_MSG();
214 
215  flush_samples.pop_back(); // Don't check the last set for equality.
216  for (size_t query_size : flush_samples) {
217  FAIL_IF(query_size != first_size) && FAIL_EXP(query_size) &&
218  FAIL_EXP(first_size);
219  }
220  }
221 };

226 /// \brief Verifies all queries only contain samples that are contiguous,
227 /// even if the set size is not a multiple of samples_per_query.
228 /// \ingroup LoadgenTestsBasic
229 struct TestMultiStreamContiguousRemainderQuery
230  : public SystemUnderTestAccuracy {
231  void SetUpTest(mlperf::TestScenario scenario) {
232  SystemUnderTestAccuracy::SetUpTest(4, 1, 0, scenario);
233  first_qsl_offsets_.resize(total_sample_count_, kBadQslOffset);
234 
235  auto spq = test_settings_.multi_stream_samples_per_query;
236  FAIL_IF(performance_sample_count_ % spq == 0) &&
237  FAIL_EXP(performance_sample_count_) && FAIL_EXP(spq) &&
238  BAD_TEST_MSG("There is no remainder.");
239  }
240 
241  void LoadSamplesToRamExt(
242  const std::vector<mlperf::QuerySampleIndex>& samples) override {
243  FAIL_IF(loaded_samples_.size() != samples.size()) &&
244  FAIL_MSG("Contiguous sample order is likely ambiguous.");
245  for (size_t i = 0; i < samples.size(); i++) {
246  auto& offset = first_qsl_offsets_.at(samples.at(i));
247  // Samples may be loaded into multiple slots for paddign purposes,
248  // so make sure to only index the first time a sample appears in a
249  // loaded set.
250  if (offset == kBadQslOffset) {
251  offset = i;
252  }
253  }
254  }
255 
256  void UnloadSamplesFromRamExt(
257  const std::vector<mlperf::QuerySampleIndex>& samples) override {
258  FAIL_IF(!loaded_samples_.empty()) &&
259  FAIL_MSG("Contiguous sample order is likely ambiguous.");
260  for (size_t i = 0; i < samples.size(); i++) {
261  first_qsl_offsets_.at(samples.at(i)) = kBadQslOffset;
262  }
263  }
264 
265  void IssueQueryExt(const std::vector<mlperf::QuerySample>& samples) override {
266  size_t expected_offset = first_qsl_offsets_[samples[0].index];
267  for (auto s : samples) {
268  FAIL_IF(loaded_samples_[expected_offset] != s.index) &&
269  FAIL_MSG("Samples are not contiguous.");
270  expected_offset++;
271  }
272  }
273 
274  void FlushQueriesExt() override {}
275 
276  void EndTest() override {}
277 
278  private:
279  static const size_t kBadQslOffset;
280  std::vector<size_t> first_qsl_offsets_;
281 };
