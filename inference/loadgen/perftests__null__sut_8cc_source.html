136 /// \brief A SUT implementation that completes queries asynchronously using
137 /// an explicitly managed thread pool.
138 class SystemUnderTestNullPool : public mlperf::SystemUnderTest {
139  public:
140  SystemUnderTestNullPool() {
141  samples_.reserve(kReserveSampleSize);
142  next_poll_time_ = std::chrono::high_resolution_clock::now() + poll_period_;
143  for (size_t i = 0; i < thread_count_; i++) {
144  threads_.emplace_back(&SystemUnderTestNullPool::WorkerThread, this);
145  }
146  }
147 
148  ~SystemUnderTestNullPool() override {
149  {
150  std::unique_lock<std::mutex> lock(mutex_);
151  keep_workers_alive_ = false;
152  }
153  cv_.notify_all();
154  for (auto& thread : threads_) {
155  thread.join();
156  }
157  }
158 
159  const std::string& Name() const override { return name_; }
160 
161  void IssueQuery(const std::vector<mlperf::QuerySample>& samples) override {
162  std::unique_lock<std::mutex> lock(mutex_);
163  samples_.insert(samples_.end(), samples.begin(), samples.end());
164  }
165 
166  void FlushQueries() override {}
167  void ReportLatencyResults(
168  const std::vector<mlperf::QuerySampleLatency>& latencies_ns) override {}
169 
170  private:
171  void WorkerThread() {
172  std::vector<mlperf::QuerySample> my_samples;
173  my_samples.reserve(kReserveSampleSize);
174  std::unique_lock<std::mutex> lock(mutex_);
175  while (keep_workers_alive_) {
176  next_poll_time_ += poll_period_;
177  auto my_wakeup_time = next_poll_time_;
178  cv_.wait_until(lock, my_wakeup_time,
179  [&]() { return !keep_workers_alive_; });
180  my_samples.swap(samples_);
181  lock.unlock();
182 
183  std::vector<mlperf::QuerySampleResponse> responses;
184  responses.reserve(my_samples.size());
185  for (auto s : my_samples) {
186  responses.push_back({s.id, 0, 0});
187  }
188  mlperf::QuerySamplesComplete(responses.data(), responses.size());
189 
190  lock.lock();
191  my_samples.clear();
192  }
193  }
194 
195  static constexpr size_t kReserveSampleSize = 1024 * 1024;
196  const std::string name_{"NullPool"};
197  const size_t thread_count_ = 4;
198  const std::chrono::milliseconds poll_period_{1};
199  std::chrono::high_resolution_clock::time_point next_poll_time_;
200 
201  std::mutex mutex_;
202  std::condition_variable cv_;
203  bool keep_workers_alive_ = true;
204  std::vector<std::thread> threads_;
205 
206  std::vector<mlperf::QuerySample> samples_;
207 };
